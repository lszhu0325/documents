# 递归创建目录
mkdir -p 目录1 目录2
# 删除文件或者文件夹
rm -rf 文件1 文件2
# 复制文件或文件夹到指定目录
cp -r 文件或文件夹 指定目录
# 移动文件或文件夹
mv 文件或文件夹 文件或文件夹
# 查看文件
ls -l
# 创建一个空白文件
touch 文件名
# 压缩文件
tar -czvf 文件名.tar.gz 源文件
# 列出压缩文件内容
tar -tzvf 压缩文件名
# 解压文件
tar -xzvf 压缩文件名
# cat命令只是将文件内容打印到标准输出上, 查看文件不建议使用
cat -n 文件名
# 显示动态文件内容
tail -f filename
# 查看文件编辑文件使用命令
vi 文件 --此模式为文件查看模式
i --此模式为编辑模式, 可进行编辑操作
: --此模式为命令行模式, 此模式可执行搜索, 粘贴复制,文件保存退出工作
# 下载文件
wget 下载地址
# 远端管理和端口探测
telnet ip:端口
# 退出当前用户
logout
# 使用新的用户登录
login
# 关机后重新开机
shutdown -r now
# 关机后停机
shutdown -h now
# 设定文件权限
chmod 权限 文件名
